package main

import (
	"fmt"
	"slices"
)

func main() {

	//Slices are dynamic, flexible arrays
	//Same as array but slices do not have fixed length
	//Slices are references to underlying arrays

	//var sliceName[]ElementType
	// var numbers []int

	// var numbers1 = []int{1, 2, 3}

	// numbers2 := []int{9, 8, 7}

	// //Initialize the slice
	// slice := make([]int, 5)

	a := [5]int{1, 2, 3, 4, 5}
	slice1 := a[1:4]

	fmt.Println(slice1)

	slice1 = append(slice1, 6, 7)
	fmt.Println("Slice1 : ", slice1)

	sliceCopy := make([]int, len(slice1))
	copy(sliceCopy, slice1)
	fmt.Println("Slicecopy : ", sliceCopy)

	// slice has zero value and do not reference anything
	// var nilSlice []int

	for i, v := range slice1 {
		fmt.Println(i, v)
	}

	fmt.Println("Element at index 3 of slice1", slice1[3])
	slice1[3] = 50
	fmt.Println("Element at index 3 of slice1", slice1[3])

	if slices.Equal(slice1, sliceCopy) {
		fmt.Println("slice1 is equal to sliceCopy")
	}

	//Multi-dimensional array/slices
	twoD := make([][]int, 3)
	for i := 0; i < 3; i++ {
		innerLen := i + 1
		twoD[i] = make([]int, innerLen)
		for j := 0; j < innerLen; j++ {
			twoD[i][j] = i + j
			fmt.Printf("Adding value %d in outer slice index %d, and in inner slice index of %d\n", i+j, i, j)
		}
	}

	fmt.Println(twoD)

	//slice operator
	// slice[low : high]
	slice2 := slice1[2:4]
	fmt.Println(slice2)

	// a slice once initialized is always associated with an underlying array
	//The capacity of a slice is the number of elements in the underlying array, counting from the first element in the slice
	//The length of a slice is the number of elements it contains
	fmt.Println("The capacity of slice2 is ", cap(slice2))
	fmt.Println("The length of slice2 is ", len(slice2))

}
