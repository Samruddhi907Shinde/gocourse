package main

import (
	"fmt"
	"unicode/utf8"
)

func main() {

	//String - sequence of bytes represents text. ITt is an array of characters
	//Strings are immutable - once ceated its value cannot be changed
	//Backtick ` ` for raw string literals
	//Rune is an integer value that represents a character

	message := "Hello, \nGo!"
	message1 := "Hello, \tGo!"
	message2 := "Hello, \rGo!" // Go!lo
	rawMessage := `Hello\nGo`

	fmt.Println(message)
	fmt.Println(message1)
	fmt.Println(message2)
	fmt.Println(rawMessage)

	fmt.Println("Length of message variable is", len(message))

	fmt.Println("The first character in message var is", message[0]) // ASCII value of H

	// No by default space in case of concatenation
	greeting := "Hello "
	name := "Alice"
	fmt.Println(greeting + name)

	str1 := "Apple"          // A has an ASCII value of 65
	str := "apple"           // a has an ASCII value of 97
	str2 := "banana"         // b has an ASCII value of 98
	str3 := "app"            // a has an ASCII value of 97
	fmt.Println(str1 < str2) // lexicographical comparison - dictionary ordering
	// ASCII value of the characters is compared
	fmt.Println(str3 < str1)
	fmt.Println(str > str1)
	fmt.Println(str > str3)

	for _, char := range message {
		// fmt.Printf("Character at index %d is %c\n", i, char)

		//hexadecimal value of the rune
		fmt.Printf("%v\n", char)
	}

	fmt.Println("Rune count:", utf8.RuneCountInString(greeting))

	greetingWithName := greeting + name
	fmt.Println(greetingWithName)

	var ch rune = 'a'
	jch := 'æ—¥'

	fmt.Println(ch)
	fmt.Println(jch)

	fmt.Printf("%c\n", ch)
	fmt.Printf("%c\n", jch)

	//Rune to string conversion
	cstr := string(ch)
	fmt.Printf("Type of cstr is %T\n", cstr)

	// go supports internationalization and multilingual text processing
	const NIHONGO = "æ—¥æœ¬èª" // Japanese text
	fmt.Println(NIHONGO)

	jhello := "ã“ã‚“ã«ã¡ã¯" // "Hello" in Japanese
	for _, runeValue := range jhello {
		fmt.Printf("%c\n", runeValue)
	}

	r := 'ğŸ˜Š'
	fmt.Printf("%v\n", r) // default value
	fmt.Printf("%c\n", r) // actual character value

}
