// Custom Errors
type myError struct {
	message string
}

// Error() is capital - error interface has a single method Error() which returns a string
// We implemented the error interface from the builtin package
// Builtin package does not need to be imported into our codebase it is imported by default and is lightweight and small package
func (m *myError) Error() string {
	return fmt.Sprintf("Error: %s", m.message)
}

func eprocess() error {
	return &myError{
		message: "Custom error message",
	}
}
