package main

import "fmt"

func main() {

	// Printing Functions
	fmt.Print("Hello ")
	fmt.Print("World!")
	fmt.Print(12, 456)

	fmt.Println("Hello ")
	fmt.Println("World!")
	fmt.Println(12, 456)

	name := "John"
	age := 25
	fmt.Printf("Name: %s, Age: %d\n", name, age)
	fmt.Printf("Binary: %b, Hex: %x\n", age, age)

	// Formatting Functions
	s := fmt.Sprint("Hello", "World!", 123, 456) // do not have newline character at the end
	fmt.Print(s)

	s = fmt.Sprintln("Hello", "World!", 123, 456) // have newline character at the end
	fmt.Print(s)
	fmt.Print(s)

	sf := fmt.Sprintf("Name: %s, Age: %d", name, age) // do not have newline character at the end
	fmt.Println(sf)
	fmt.Println(sf)

	//Scanning Functions - scans text from standard input and stores it into variables
	var name1 string
	var age1 int

	fmt.Print("Enter your and age:")
	// Scan will keep asking for correct input even if you press enter
	// Scanln will stop execution as soon as it encounters enter( new line )
	// fmt.Scan(&name1, &age1) // & - memory address to modify the original variables
	// fmt.Scanln(&name1, &age1)
	fmt.Scanf("%s %d", &name1, &age1)
	fmt.Printf("Name: %s, Age: %d\n", name1, age1)

	// Error Formatting Functions
	err := checkAge(15)
	if err != nil {
		fmt.Println("Error:", err)
	}

}

func checkAge(age int) error {
	if age < 18 {
		return fmt.Errorf("Age %d is too young to drive.", age)
	}
	return nil
}
