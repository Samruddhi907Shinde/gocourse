package main

import (
	"fmt"
	"math"
)

// Interfaces are also declared outside the main function like structs and methods
// A type implicitly satisfies an interface if it implements all the methods defined by that interface
// Interface promotes - code reuse, decoupling, and polymorphism
// Any method or function (type) needs to be exported should start with an uppercase
// All the methods defined in an interface should be implemented by that struct to be able to get associated with that interface type
// Additional methods are okay

type geometry interface {
	// these two are undefined functions
	area() float64
	perim() float64
}

type rect struct {
	width, height float64
}

// The return value was already defined in the interface
// Here we are configuring the internal mechanism of the area function
func (r rect) area() float64 {
	return r.width * r.height
}

func (r rect) perim() float64 {
	return 2 * (r.width + r.height)
}

type circle struct {
	radius float64
}

func (c circle) area() float64 {
	return math.Pi * c.radius * c.radius
}

func (c circle) perim() float64 {
	return 2 * math.Pi * c.radius
}

func (c circle) diameter() float64 {
	return 2 * c.radius
}

type rect1 struct {
	width, height float64
}

func (r rect1) area() float64 {
	return r.width * r.height
}

// func (r rect1) perim() float64 {

// Common function to print output for all shapes of geometry type
func measure(g geometry) {
	fmt.Println(g)
	fmt.Println(g.area())
	fmt.Println(g.perim())
}

func main() {

	// In go interfaces provide a way to specify behaviours
	r := rect{width: 3, height: 4}
	c := circle{radius: 5}
	// r1 := rect1{width: 3, height: 4}

	measure(r)
	measure(c)
	// measure(r1)

	myPrinter(1, "John", 45.9, true)

	printType(9)
	printType("Hello")
	printType(false)

}

// Accepting any no of parameters in my function with any type
func myPrinter(i ...interface{}) {
	for _, v := range i {
		fmt.Println(v)
	}
}

// Function to accept any type of value
// Interface means flexible to enter any type of value
func printType(i interface{}) {
	switch i.(type) {
	case int:
		fmt.Println("Type: Int")
	case string:
		fmt.Println("Type: String")
	default:
		fmt.Println("Type: Unknown")
	}
}
