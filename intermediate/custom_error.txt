package main

import (
	"errors"
	"fmt"
)

type customError struct {
	code    int
	message string
	er      error
}

// Error returns the error message. Implementing Error() method returning string of error interface
func (e *customError) Error() string {
	return fmt.Sprintf("Error %d: %s, %v\n", e.code, e.message, e.er)
}

	// Custom Errors ensure consistency in error handling accross our codebase.
