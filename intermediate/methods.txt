package main

import "fmt"

type Shape struct {
	Rectangle
}
type Rectangle struct {
	length float64
	width  float64
}

// Method with value receiver - do not modify receiver instance
// Just like instnace of the class, we get the list of properties and methods
func (r Rectangle) Area() float64 {
	return r.length * r.width
}

// Method with pointer receiver- modifies the receiver instance or to avoid copying large structs
func (r *Rectangle) Scale(factor float64) {
	r.length *= factor
	r.width *= factor
}

// Methods can be associated with any type not only with structs
type MyInt int

// Method on a user-defined type
func (m MyInt) IsPositive() bool {
	return m > 0
}

// No need to create instance m because we are not accessing any value
// We are suppose to use an instance only for extracting or modifying a value
func (MyInt) welcomeMessage() string {
	return "Welcome to MyInt Type"
}

func main() {

	rect := Rectangle{length: 10, width: 9}
	area := rect.Area()
	fmt.Println("Area of Rectangle with length 10 and width 9 is ", area)
	rect.Scale(2)
	area = rect.Area()
	fmt.Println("Area of Rectangle after scaling by 2 is ", area)

	num := MyInt(-5)
	num1 := MyInt(9)
	fmt.Println(num.IsPositive())
	fmt.Println(num1.IsPositive())
	fmt.Println(num.welcomeMessage())

	s := Shape{
		Rectangle: Rectangle{
			length: 10, width: 9,
		},
	}
	fmt.Println(s.Area())
	fmt.Println(s.Rectangle.Area())

}
